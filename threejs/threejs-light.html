    function createLight(color) {

  const intensity = 2;

  const light = new THREE.PointLight(color, intensity, 20);

  // å¼€å¯é˜´å½±

  light.shadow.bias = -0.005;

  // åˆ›å»ºä¸€ä¸ªçƒä½“ä½œä¸ºå…‰æºçš„å¯è§†åŒ–è¡¨ç¤º
  let geometry = new THREE.SphereGeometry(0.3, 12, 6);

    color: color
  });

  // æ ¹æ®å…‰æºå¼ºåº¦è°ƒæ•´é¢œè‰²
  material.color.multiplyScalar(intensity);

  let sphere = new THREE.Mesh(geometry, material);

  // å°†çƒä½“æ·»åŠ åˆ°å…‰æºä¸Š
  light.add(sphere);

  // è®¾ç½®çº¹ç†è¿‡æ»¤å™¨å’Œé‡å¤æ–¹å¼
  texture.magFilter = THREE.NearestFilter;

  // åˆ›å»ºä¸€ä¸ªçƒä½“ä½œä¸ºå…‰æºçš„å…‰æ™•æ•ˆæœ
  geometry = new THREE.SphereGeometry(2, 32, 8);

    alphaMap: texture, // è®¾ç½®é€æ˜åº¦è´´å›¾
    alphaTest: 0.5 // è®¾ç½®é€æ˜åº¦é˜ˆå€¼

  sphere = new THREE.Mesh(geometry, material);

  // å¼€å¯çƒä½“é˜´å½±å’Œæ¥æ”¶é˜´å½±
  sphere.castShadow = true;
  sphere.receiveShadow = true;

  // å°†çƒä½“æ·»åŠ åˆ°å…‰æºä¸Š
  light.add(sphere);
