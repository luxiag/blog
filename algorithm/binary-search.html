function minimumBoxes(n: number): number {
    let i = 0, j = 0;
    let low = 1, high = Math.min(n, 100000);
    while (low < high) {
        const mid = Math.floor((low + high) / 2);
        const sum = (mid * (mid + 1) * (mid + 2)) / 6;
        if (sum >= n) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    i = low;
    n -= Math.floor((i - 1) * i * (i + 1)) / 6;
    low = 1;
    high = i;
    while (low < high) {
        const mid = Math.floor((low + high) / 2);
        const sum = (mid * (mid + 1)) / 2;
        if (sum >= n) {
            high = mid;
        } else {
            low = mid + 1;
        }
    }
    j = low;
    return Math.floor((i - 1) * i / 2) + j;
